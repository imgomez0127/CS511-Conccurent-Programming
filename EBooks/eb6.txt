monitor TWS{
    int state = 1;
    condition first,second,third;
    
    first(){
        while(state != 1){
            first.wait();
        }
        state = 2;
        second.signal();
    }
    second(){
        while(state !=2){
            second.wait();
        }
        state = 3;
        third.signal()
    }
    third(){
        while(state != 3){
            third.wait();
        }
        state = 1;
        first.signal()
    }
}

monitor Barrier(n){
    int amount_in_barrier = 0 
    condition wait_bitch
    synchronize(){
        if amount_in_barrier <= n
            amount_in_barrier++
            while amount_in_barrier < n-1:
                wait_bitch.wait()
            if amount_in_barrier == n-1:
                for i=0; i < n-1; ++i:
                    wait_bitch.signal()
    }
}
